#Set permissions for GMs (No world-altering commands included) 



UPDATE command
SET security = CASE name 
						WHEN 'character rename' THEN 1
						WHEN 'npc say' THEN 1
						WHEN 'npc yell' THEN 1
						WHEN 'baninfo account' THEN 1
						WHEN 'baninfo character' THEN 1
						WHEN 'account characters' THEN 1
						ELSE name
						END
WHERE name IN('character rename', 'npc say', 'npc yell', 'baninfo account','baninfo character','account characters');  
						





UPDATE command
SET security=1
WHERE name LIKE 'lookup%';
